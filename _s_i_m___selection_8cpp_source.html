226 void SIM_Selection::bimSelectionChanged(const QString &arg1)
227 {
228  selectionChangeOK = true;
229 
230  if (bimInput != 0)
231  layout->removeWidget(bimInput);
232 
233  //
234  // note type output in json and name in pull down are not the same and hence the ||
235  //
236 
237 
238 
239  if (arg1 == QString("OpenSees") || (arg1 == QString("OpenSeesInput"))) {
240 
241  delete bimInput;
242  bimInput = new OpenSeesBuildingModel(theRandomVariablesContainer, includeCentroid);
243  } else if (arg1 == QString("MDOF") || (arg1 == QString("MDOF_BuildingModel"))) {
244  delete bimInput;
245  bimInput = new MDOF_BuildingModel(theRandomVariablesContainer);
246  } else {
247  selectionChangeOK = false;
248  emit sendErrorMessage("ERROR: BIM Input - no valid Method provided .. keeping old");
249  }
250 
251  if (bimInput != 0) {
252  this->bimWidgetChanged();
253  layout->insertWidget(1, bimInput,1);
254  connect(bimInput,SIGNAL(sendErrorMessage(QString)),this,SLOT(errorMessage(QString)));
255  }
256 
257  return;
258 }
