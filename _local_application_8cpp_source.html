171  emit sendErrorMessage(QString("NO REGISTRY FILE: ") + registryFile);
172  return false;
173  }
174 
175  qDebug() << "SCRIPT: " << pySCRIPT;

177 
178  QStringList files;
179  files << "dakota.in" << "dakota.out" << "dakotaTab.out" << "dakota.err";
180 
181  //emit sendStatusMessage("Running Dakota .. either run remotely or patience!");

183  qDebug() << "Running the UQ engine ... ";
184  emit sendStatusMessage("Running the UQ engine ... ");

188  //

191

194  QString pathEnv = procEnv.value("PATH");
195  QString pythonPathEnv = procEnv.value("PYTHONPATH");
196 
197

199  QString exportPath("export PATH=$PATH");
200 
201  QSettings settings("SimCenter", "Common"); //These names will need to be constants to be shared
202  QVariant pythonLocationVariant = settings.value("pythonExePath");
203  if (pythonLocationVariant.isValid()) {
204  python = pythonLocationVariant.toString();
205  }
206 
207  QSettings settingsApplication("SimCenter", QCoreApplication::applicationName());
208  QVariant openseesPathVariant = settingsApplication.value("openseesPath");
209  if (openseesPathVariant.isValid()) {
210  QFileInfo openseesFile(openseesPathVariant.toString());
211  if (openseesFile.exists()) {
212  QString openseesPath = openseesFile.absolutePath();
213  pathEnv = openseesPath + ';' + pathEnv;
214  exportPath += ":" + openseesPath;
215  }
216  }
217 
218  QVariant dakotaPathVariant = settingsApplication.value("dakotaPath");
219  if (dakotaPathVariant.isValid()) {
220  QFileInfo dakotaFile(dakotaPathVariant.toString());
221  if (dakotaFile.exists()) {
222  QString dakotaPath = dakotaFile.absolutePath();
223  QString dakotaPythonPath = QFileInfo(dakotaPath).absolutePath() + QDir::separator() +
224  "share" + QDir::separator() + "Dakota" + QDir::separator() + "Python";
225  exportPath += ":" + dakotaPath;
226  pathEnv = dakotaPath + ';' + pathEnv;
227  pythonPathEnv = dakotaPythonPath + ";" + pythonPathEnv;
228  }
229  }
230 
231  procEnv.insert("PATH", pathEnv);
232  procEnv.insert("PYTHONPATH", pythonPathEnv);
233  proc->setProcessEnvironment(procEnv);
234 
235  qDebug() << "PATH: " << pathEnv;
236  qDebug() << "PYTHON_PATH" << pythonPathEnv;
237 
238  QStringList args{pySCRIPT, runType, inputFile, registryFile};
239 
240 #ifdef Q_OS_WIN
241  python = QString("\"") + python + QString("\"");
242 
243  qDebug() << python;
244  qDebug() << args;
245 
246  proc->start(python,args);
247 
248  bool failed = false;
249  if (!proc->waitForStarted(-1))
250  {
251  qDebug() << "Failed to start the workflow!!! exit code returned: " << proc->exitCode();
252  qDebug() << proc->errorString().split('\n');
253  emit sendStatusMessage("Failed to start the workflow!!!");
254  failed = true;
255  }
256 
257  if(!proc->waitForFinished(-1))
258  {
259  qDebug() << "Failed to finish running the workflow!!! exit code returned: " << proc->exitCode();
260  qDebug() << proc->errorString();
261  emit sendStatusMessage("Failed to finish running the workflow!!!");
262  failed = true;
263  }
264 
265 
266  if(0 != proc->exitCode())
267  {
268  qDebug() << "Failed to run the workflow!!! exit code returned: " << proc->exitCode();
269  qDebug() << proc->errorString();
270  emit sendStatusMessage("Failed to run the workflow!!!");
271  failed = true;
272  }
273 
274  if(failed)
275  {
276  qDebug().noquote() << proc->readAllStandardOutput();
277  qDebug().noquote() << proc->readAllStandardError();
278  return false;
279  }
280 
281 #else
282 
283  // check for bashrc or bash profile
284  QDir homeDir(QDir::homePath());
285  QString sourceBash("\"");
286  if (homeDir.exists(".bash_profile")) {
287  sourceBash = QString("source $HOME/.bash_profile; ");
288  } else if (homeDir.exists(".bashrc")) {
289  sourceBash = QString("source $HOME/.bashrc; ");
290  } else if (homeDir.exists(".zprofile")) {
291  sourceBash = QString("source $HOME/.zprofile; ");
292  } else if (homeDir.exists(".zshrc")) {
293  sourceBash = QString("source $HOME/.zshrc; ");
294  } else
295  emit sendErrorMessage( "No .bash_profile, .bashrc or .zshrc file found. This may not find Dakota or OpenSees");
296 
297  // note the above not working under linux because bash_profile not being called so no env variables!!
298  QString command = sourceBash + exportPath + "; \"" + python + QString("\" \"" ) +
299  pySCRIPT + QString("\" " ) + runType + QString(" \"" ) + inputFile + QString("\" \"") + registryFile + QString("\"");
300 
301  qDebug() << "PYTHON COMMAND" << command;
302 
303  proc->execute("bash", QStringList() << "-c" << command);
304  proc->waitForStarted();
305 
306 #endif

326  return 0;
