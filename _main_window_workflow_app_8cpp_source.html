389 {

394  QFile file(fileName);
395  if (!file.open(QFile::WriteOnly | QFile::Text)) {
396  QMessageBox::warning(this, tr("Application"),

429 {
430  //
431  // open file
432  //
433 
434  QFile file(fileName);
435  if (!file.open(QFile::ReadOnly | QFile::Text)) {
436  QMessageBox::warning(this, tr("Application"),
437  tr("Cannot read file %1:\n%2.")
438  .arg(QDir::toNativeSeparators(fileName), file.errorString()));
439  return;
440  }
441 
442  // validate the document
443  // JsonValidator class already takes a model type param, add additional model types as required
444 
445  /*
446  JsonValidator *jval = new JsonValidator();

448 */
449 
450  // place contents of file into json object
451  QString val;
452  val=file.readAll();
453  QJsonDocument doc = QJsonDocument::fromJson(val.toUtf8());
454  QJsonObject jsonObj = doc.object();
455 
456  //
457  QFileInfo fileInfo(fileName);
458  SCUtils::ResolveAbsolutePaths(jsonObj, fileInfo.dir());
459 
460  // close file
461  file.close();
462 
463  // given the json object, create the C++ objects
464  if ( ! (currentFile.isNull() || currentFile.isEmpty()) ) {
465  inputWidget->clear();
466  }
467  inputWidget->inputFromJSON(jsonObj);
468 
469  setCurrentFile(fileName);
