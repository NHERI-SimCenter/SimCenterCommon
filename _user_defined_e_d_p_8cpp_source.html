349 bool
350 UserDefinedEDP::setProcessingScript(QString filename){
351 
352  // set file name & ebtry in qLine edit
353  //filenameProcessingScript = name;
354 
355  // remove old EDP
356  int numEDPs = theEDPs.size();
357  for (int i = numEDPs-1; i >= 0; i--) {
358  EDP *theEDP = theEDPs.at(i);
359  theEDP->close();
360  edpLayout->removeWidget(theEDP);
361  theEDPs.remove(i);
362  delete theEDP;
363  }
364 
365  // set filename
366  processingScriptLE->setText(filename);
367 
368  // process file looking for line with EDPs
369  std::ifstream in_file(filename.toStdString());
370  unsigned int max_iters = 100;
371  unsigned int count = 0;
372  bool found_line = false;
373  std::string current_line;
374  std::string EDP_array("EDPs");
375  std::string token;
376 
377  // Iterate over file until line containing EDPs is found or maximum number of iterations
378  // is reached
379  while (!found_line && count < max_iters) {
380  // Get current line and search for EDP array
381  std::getline(in_file, current_line);
382  std::size_t found_EDPs = current_line.find(EDP_array);
383 
384  // If EDP array found, iterate over it to find EDPs and add them
385  if (found_EDPs != std::string::npos) {
386  std::stringstream line_to_check(current_line);
387  unsigned int counter = 0;
388  while (std::getline(line_to_check, token, '\'')) {
389  counter++;
390  // Since names are between two "'" characters, only add token after second occurance
391  if (counter == 2) {
392  QString name = QString::fromStdString(token);
393  this->addEDP(name);
394  counter = 0;
395  }
396  }
397  found_line = true;
398  }
399  count++;
400  }
401 
402  // close file
403  in_file.close();
404 
405  if(!found_line)
406  return false;
407 
408  return true;
409 }
