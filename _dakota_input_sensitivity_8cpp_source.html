172 DakotaInputSensitivity::inputFromJSON(QJsonObject &jsonObject)
173 {
174  bool result = false;
175  this->clear();
176 
177  //
178  // get sampleingMethodData, if not present it's an error
179  //
180 
181  if (jsonObject.contains("samplingMethodData")) {
182  QJsonObject uq = jsonObject["samplingMethodData"].toObject();
183  if (uq.contains("method")) {
184  QString method =uq["method"].toString();
185  int index = samplingMethod->findText(method);
186  if (index == -1) {
187  return false;
188  }
189  samplingMethod->setCurrentIndex(index);
190  result = theCurrentMethod->inputFromJSON(uq);
191  if (result == false)
192  return result;
193  }
194  }
195 
196  return result;
197 }
