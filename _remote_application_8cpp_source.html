321  //
322  // create the json needed to run the remote application

324 
325  QJsonObject job;

327  pushButton->setDisabled(true);

329  job["name"]=shortDirName + nameLineEdit->text();

346  parameters["errorFile"]="dakota.err";

348  parameters["modules"]="petsc,python3";
349  for (auto parameterName : extraParameters.keys())
350  {
351  parameters[parameterName] = extraParameters[parameterName];
352  }
353  job["parameters"]=parameters;
354 
355  QDir theDirectory(tempDirectory);
356  QString dirName = theDirectory.dirName();
357 
358  QString remoteDirectory = remoteHomeDirPath + QString("/") + dirName;
359 
360  QJsonObject inputs;
361 
362  inputs["inputDirectory"]=remoteDirectory;
363  for (auto inputName : extraInputs.keys())
364  {
365  inputs[inputName] = extraInputs[inputName];
366  }
367  job["inputs"]=inputs;
368 
369  // disable the button while the job is being uploaded and started

372  qDebug() << "JOBS_SUBMIT: " << job;
