104 
105  if (theSpreadsheet->getString(i,2, mass) == false)
106  break;

109  mass.replace("[", "");
110  mass.replace("]", "");
111  mass.replace(",", " ");
112 
113  QJsonArray massArray;
114  istringstream iss(mass.toStdString());
115  do
116  {
117  std::string subs;

119  massArray.append(std::atof(subs.c_str()));
120  } while (iss);
121 
122 
123  // now add the items to object
124  obj["name"]=name;
125  obj["type"]=type;
126  obj["mass"]=massArray;
127 
128  // add the object to the array
129  jsonArray.append(obj);

131  }
