208 BraceInputWidget::inputFromJSON(QJsonObject &jsonObject){
209 
210 
211  QString name;
212  QJsonArray clineArray;
213  QJsonArray floorArray;
214  QString cline1Value, cline2Value, floor1Value, floor2Value;
215  int currentRow = 0;
216 
217  //
218  // get the cline data (a json array) from the object, and for every
219  // object in the array, get the values and add to the spreadsheet
220  //
221 
222  QJsonArray theArray = jsonObject["braces"].toArray();
223  foreach (const QJsonValue &theValue, theArray) {
224  // get values
225  QJsonObject theObject = theValue.toObject();
226  QJsonValue theColumnValue = theObject["name"];
227  name = theColumnValue.toString();
228 
229  QJsonValue theClineValue = theObject["cline"];
230  clineArray = theClineValue.toArray();
231  QJsonValue c1Value = clineArray.at(0);
232  QJsonValue c2Value = clineArray.at(1);
233  cline1Value = c1Value.toString();
234  cline2Value = c2Value.toString();
235 
236  QJsonValue theFloorValue = theObject["floor"];
237  floorArray = theFloorValue.toArray();
238  QJsonValue f1Value = floorArray.at(0);
239  QJsonValue f2Value = floorArray.at(1);
240  floor1Value = f1Value.toString();
241  floor2Value = f2Value.toString();
242 
243  QJsonArray theSegmentArray = theObject["segment"].toArray();
244 
245  // add to the spreadsheet
246  theSpreadsheet->setString(currentRow, 0, name);
247  theSpreadsheet->setString(currentRow, 1, floor1Value);
248  theSpreadsheet->setString(currentRow, 2, floor2Value);
249  theSpreadsheet->setString(currentRow, 3, cline1Value);
250  theSpreadsheet->setString(currentRow, 4, cline2Value);
251 
252 
253  QString section;
254  double angle;
255  double ratio1Value, ratio2Value;
256  int currentSection = 0;
257  int offset = 0;
258 
259  foreach (const QJsonValue &theValue, theSegmentArray) {
260 
261  // get values
262  QJsonObject theObject = theValue.toObject();
263 
264  section = theObject["section"].toString();
265  angle = theObject["angle"].toDouble();
266 
267  // ratio can be a two member array or single value
268  if ( theObject["ratio"].isArray() ) {
269  QJsonArray theRatioArray = theObject["ratio"].toArray();
270  QJsonValue r1Value = theRatioArray.at(0);
271  QJsonValue r2Value = theRatioArray.at(1);
272  ratio1Value = r1Value.toDouble();
273  ratio2Value = r2Value.toDouble();
274  }
275  else {
276  QJsonValue r1Value = theObject["ratio"];
277  ratio1Value = r1Value.toDouble();
278  // TODO: correect default for ratio2?????
279  ratio2Value = 0.0;
280  }
281 
282  // all segment sections on the same row so must shift for each new set
283  offset = (currentSection * 4) + 5;
284  theSpreadsheet->setString(currentRow, offset, section);
285  theSpreadsheet->setDouble(currentRow, (offset + 1), ratio1Value);
286  theSpreadsheet->setDouble(currentRow, (offset + 2), ratio2Value);
287  theSpreadsheet->setDouble(currentRow, (offset + 3), angle);
288 
289  currentSection++;
290  }
291  currentRow++;
292  }
293 
294  return(true);
295 }
