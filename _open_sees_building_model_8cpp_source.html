151 OpenSeesBuildingModel::outputToJSON(QJsonObject &jsonObject)
152 {
153  // just need to send the class type here.. type needed in object in case user screws up
154  jsonObject["type"]="OpenSeesInput";
155  QJsonArray nodeTags;
156  if (centroidNodes != NULL) {
157  string nodeString = centroidNodes->text().toStdString();
158  string s1(nodeString); // this line is needed as nodeString cannot be passed directly to the line below!
159  stringstream nodeStream(s1);
160  int nodeTag;
161  while (nodeStream >> nodeTag) {
162  nodeTags.append(QJsonValue(nodeTag));
163  if (nodeStream.peek() == ',')
164  nodeStream.ignore();
165  }
166  }
167 
168  QJsonArray responseNodeTags;
169  string nodeString = responseNodes->text().toStdString();
170  string s1(nodeString); // this line is needed as nodeString cannot be passed directly to the line below!
171  stringstream nodeStream(s1);
172  int nodeTag;
173  int numStories = -1;
174  while (nodeStream >> nodeTag) {
175  responseNodeTags.append(QJsonValue(nodeTag));
176  if (nodeStream.peek() == ',') {
177  nodeStream.ignore();
178  } else
179  numStories++;
180  }
181 
182  jsonObject["centroidNodes"]=nodeTags;
183  jsonObject["responseNodes"]=responseNodeTags;
184  jsonObject["ndm"]=ndm->text().toInt();
185  jsonObject["ndf"]=ndf->text().toInt();
186  //jsonObject["stories"]=numStories;
187 
188  QJsonArray rvArray;
189  for (int i=0; i<varNamesAndValues.size()-1; i+=2) {
190  QJsonObject rvObject;
191  QString name = varNamesAndValues.at(i);
192  rvObject["name"]=name;
193  rvObject["value"]=QString("RV.")+name;
194  rvArray.append(rvObject);
195  }
196 
197  jsonObject["randomVar"]=rvArray;
198 
199  return true;
200 }

203 bool
204 OpenSeesBuildingModel::inputFromJSON(QJsonObject &jsonObject)
205 {
206  varNamesAndValues.clear();
207 
208  this->clear();
209  QString stringNodes;
210  if (jsonObject.contains("centroidNodes")) {
211  QJsonArray nodeTags = jsonObject["centroidNodes"].toArray();
212  int numCentroid = 0;
213  foreach (const QJsonValue & value, nodeTags) {
214  int tag = value.toInt();
215  stringNodes = stringNodes + " " + QString::number(tag);
216  numCentroid++;
217  }
218  if (numCentroid != 0) {
219  if (centroidNodes == NULL) {
220  QLabel *label2a = new QLabel();
221  label2a->setText("Centroid Nodes:");
222  centroidNodes = new QLineEdit;
223  layout->addWidget(label2a,4,0);
224  layout->addWidget(centroidNodes,4,1);
225  }
226  centroidNodes->setText(stringNodes);
227  }
228  }
229 
230  if (jsonObject.contains("responseNodes")) {
231  QString stringResponseNodes;
232  QJsonArray nodeResponseTags = jsonObject["responseNodes"].toArray();
233  foreach (const QJsonValue & value, nodeResponseTags) {
234  int tag = value.toInt();
235  stringResponseNodes = stringResponseNodes + " " + QString::number(tag);
236  }
237  responseNodes->setText(stringResponseNodes);
238  }
239 
240  // backward compatability .. response nodes used to be nodes
241  if (jsonObject.contains("nodes")) {
242  QString stringResponseNodes;
243  QJsonArray nodeResponseTags = jsonObject["nodes"].toArray();
244  foreach (const QJsonValue & value, nodeResponseTags) {
245  int tag = value.toInt();
246  stringResponseNodes = stringResponseNodes + " " + QString::number(tag);
247  }
248  responseNodes->setText(stringResponseNodes);
249  }
250 
251  if (jsonObject.contains("randomVar")) {
252  QJsonArray randomVars = jsonObject["randomVar"].toArray();
253  foreach (const QJsonValue & value, randomVars) {
254  QJsonObject theRV = value.toObject();
255  QString name = theRV["name"].toString();
256  QString zero = "0";
257  varNamesAndValues.append(name);
258  varNamesAndValues.append(zero);
259  }
260  }
261 
262  int theNDM = jsonObject["ndm"].toInt();
263  int theNDF = 1;
264  if (theNDM == 2)
265  theNDF = 3;
266  else if (theNDM == 3)
267  theNDF = 6;
268 
269  if (jsonObject.contains("ndf")) {
270  theNDF = jsonObject["ndf"].toInt();
271  }
272 
273  ndm->setText(QString::number(theNDM));
274  ndf->setText(QString::number(theNDF));
275 
276  return true;
277 }
