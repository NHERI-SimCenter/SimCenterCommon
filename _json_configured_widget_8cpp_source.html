156 QWidget *
157 JsonConfiguredWidget::generateWidgetLayout(const QJsonArray &inputArray,
158  bool set) const {
159 
160  auto generatedWidget = new QWidget();
161  QVBoxLayout * layout = new QVBoxLayout();
162 
163  for (auto const& value : inputArray) {
164  switch (getEnumIndex(value["type"].toString())) {
165  case JsonWidget::Type::ComboBox: {
166  auto comboBox = generateComboBox(value);
167  if (set) {
168  auto jsonObject = value.toObject();
169  comboBox->inputFromJSON(jsonObject);
170  }
171  layout->addWidget(comboBox);
172  break;
173  }
174 
175  case JsonWidget::Type::RVLineEdit: {
176  auto rvLineEdit = generateRVLineEdit(value);
177  if (set) {
178  auto jsonObject = value.toObject();
179  rvLineEdit->inputFromJSON(jsonObject);
180  }
181  layout->addWidget(rvLineEdit);
182  break;
183  }
184 
185  case JsonWidget::Type::FileInput: {
186  auto fileInput = generateFileInput(value);
187  if (set) {
188  auto jsonObject = value.toObject();
189  fileInput->inputFromJSON(jsonObject);
190  }
191  layout->addWidget(fileInput);
192  break;
193  }
194  default:
195  break;
196  }
197  }
198 
199  layout->addStretch();
200  generatedWidget->setLayout(layout);
201 
202  return generatedWidget;
203 }

226
