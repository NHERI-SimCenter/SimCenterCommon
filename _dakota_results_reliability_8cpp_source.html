196  /* **************************************** LOOKING FOR THE FOLLOWING
197  -----------------------------------------------------------------
198  Cumulative Distribution Function (CDF) for response_fn_1:
199  Response Level Probability Level Reliability Index General Rel Index
200  -------------- ----------------- ----------------- -----------------
201  2.5359850842e-01 1.9999879291e-02 2.0537514037e+00 2.0537514037e+00
202  ........ ........ ....... .........
203  ........ ........ ....... .........
204  -----------------------------------------------------------------
205 
206  *************************************************************************** */
207 
208  const std::string needleStart = "Cumulative Distribution Function (CDF)";
209  std::string haystack;
210 
211  bool foundResults = false;

213  while(foundResults == false && fileResults.eof() != true) {
214 
215  std::vector<double>col1;
216  std::vector<double>col2;
217 
218  while (std::getline(fileResults, haystack)) {
219  if (haystack.find(needleStart) != std::string::npos) {
220  foundResults = true;
221  break;
222  }
223  }
224

228  }
229  }
230

233  return -1;

237  * for each response quantity enter prob levels into spreadsheet

245  do {
246  iss >> subs;
247  } while (iss);

252

261

264 
265  QMap<float,float> data;
266  // now read the data till end of data encountered
267  int numRows = 0;
268 
269  while (std::getline(fileResults, haystack)) {
270  if (haystack.find(needleStart) != std::string::npos) {
271  break;
272  } else if (haystack.find(needleEnd) != std::string::npos) {
273  qDebug() << "FOUND END";
274  endResponses = true;
275  break;
276  } else {
277  // read column entries
278  if (numSpreadsheetCols == 2)
279  spreadsheet->insertRow(numRows);
280 
281  std::string data1, data2, data3, data4;
282  std::istringstream is(haystack);
283 
284  is >> data1 >> data2 >> data3 >> data4;
285 
286  if (numSpreadsheetCols == 2) {
287  data.insert(std::stod(data2),std::stod(data1));
288  QModelIndex index = spreadsheet->model()->index(numRows, numSpreadsheetCols-2);
289  spreadsheet->model()->setData(index, data2.c_str());
290  }
291  QModelIndex index = spreadsheet->model()->index(numRows, numSpreadsheetCols-1);
292  spreadsheet->model()->setData(index, data1.c_str());
293 
294  numRows++;
295  }
296  }
297 
298  numSpreadsheetRows = numRows;
299  }
300 
301  spreadsheet->setHorizontalHeaderLabels(theHeadings);
302  spreadsheet->horizontalHeader()->setSectionResizeMode(QHeaderView::Stretch);
303 
304  fileResults.close();
305 
306  this->onSpreadsheetCellClicked(0,numSpreadsheetCols-1);
307  if (numSpreadsheetRows == 0)
308  emit sendStatusMessage(tr("No Result Data Found .. dakota failed .. possibly no QoI provided"));
309 
310  return 0;
