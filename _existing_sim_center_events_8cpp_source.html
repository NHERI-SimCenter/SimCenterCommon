229 void ExistingSimCenterEvents::loadEventsFromDir(void) {
230  QString dir = QFileDialog::getExistingDirectory(this, tr("Open Directory"),
231  "/home",
232  QFileDialog::ShowDirsOnly
233  | QFileDialog::DontResolveSymlinks);
234 
235  this->clear();
236 
237  QDir directory(dir);
238  QString recordsTxt(directory.filePath("Records.txt"));
239  QFileInfo checkFile(recordsTxt);
240 
241  if (checkFile.exists() && checkFile.isFile()) {
242 
243  QFile file(recordsTxt);
244  if (!file.open(QIODevice::ReadOnly)) {
245  qDebug() << file.errorString();
246  return;
247  }
248 
249  QStringList wordList;
250  while (!file.atEnd()) {
251  QByteArray line = file.readLine();
252  QByteArrayList lineList = line.split(',');
253  if (lineList.length() >= 2) {
254 
255  QString fileName = lineList.at(0);
256  QString factor = lineList.at(1);
257  factor.remove(QRegExp("[\\n\\t\\r]"));
258 
259  QFileInfo checkName(directory.filePath(fileName));
260  if (checkName.exists() && checkName.isFile()) {
261 
262  ExistingEvent *theEvent = new ExistingEvent(theRandVariableIW);
263  QFileInfo infoFile(fileName);
264  QString name = infoFile.baseName();
265  theEvent->theName->setText(name);
266  theEvent->file->setText(directory.filePath(fileName));
267  theEvent->factor->setText(factor);
268 
269  theEvents.append(theEvent);
270  eventLayout->insertWidget(eventLayout->count()-1, theEvent);
271 
272  } else {
273  qDebug() << "ExistingEvents: load directory file " << fileName << " does not exist";
274  }
275  }
276 
277  }
278 
279  file.close();
280 
281  } else {
282 
283  QStringList fileList= directory.entryList(QStringList() << "*.json",QDir::Files);

285  ExistingEvent *theEvent = new ExistingEvent(theRandVariableIW);
