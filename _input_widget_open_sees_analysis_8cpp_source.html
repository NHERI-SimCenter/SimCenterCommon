305

308  if (ok == true)
309  jsonObject["dampingRatio"]=dampDouble;
310  else
311  jsonObject["dampingRatio"]= QString("RV.") + dampText;
312 
313  QString dampModalText = dampingRatioModal->text();
314  double dampModalDouble = dampModalText.QString::toDouble(&ok);
315  if (ok == true)
316  jsonObject["dampingRatioModal"]=dampModalDouble;
317  else
318  jsonObject["dampingRatioModal"]= QString("RV.") + dampModalText;
319 
320 
321 
322 
323  if (!file->text().isEmpty() && !file->text().isNull()) {
324  QFileInfo fileInfo(file->text());
325  jsonObject["fileName"]= fileInfo.fileName();
326  jsonObject["filePath"]=fileInfo.path();
327  }
328 
329  return result;

334 InputWidgetOpenSeesAnalysis::inputFromJSON(QJsonObject &jsonObject)
335 {
336  bool result = true;
337  this->clear();
338 
339  if (jsonObject.contains("integration") && jsonObject.contains("algorithm")
340  && jsonObject.contains("convergenceTest")) {
341 
342  // theAlgorithm->setText(jsonObject["algorithm"].toString());
343  theAlgorithm->setCurrentText(jsonObject["algorithm"].toString());
344  theConvergenceTest->setText(jsonObject["convergenceTest"].toString());
345  theIntegration->setText(jsonObject["integration"].toString());
346 
347  } else {
348  emit sendErrorMessage("ERROR: InputWidgetOpenSeesAnalysis - no \"integration\" ,\"convergenceTest\" or \"algorithm\" data");
349  return false;
350  }
351 
352 
353  if (jsonObject.contains("dampingRatio")) {
354  QJsonValue theValue = jsonObject["dampingRatio"];
355  if (theValue.isString()) {
356  QString text = theValue.toString();
357  text.remove(0,3); // remove RV.
358  dampingRatio->setText(text);
359  } else if (theValue.isDouble())
360  dampingRatio->setText(QString::number(theValue.toDouble()));
361  } else {
362  emit sendErrorMessage("ERROR: InputWidgetOpenSeesAnalysis - no \"dampingRatio\" data");
363  return false;
364  }
365 
366  if (jsonObject.contains("dampingRatioModal")) {
367  QJsonValue theValue = jsonObject["dampingRatioModal"];
368  if (theValue.isString()) {
369  QString text = theValue.toString();
370  text.remove(0,3); // remove RV.
371  dampingRatioModal->setText(text);
372  } else if (theValue.isDouble())
373  dampingRatioModal->setText(QString::number(theValue.toDouble()));
374  } else {
375  dampingRatioModal->setText("0.02");// old code, use defaults
376  }
377 
378  if (jsonObject.contains("dampingModel")) {
379  QJsonValue theValue = jsonObject["dampingModel"];
380  if (theValue.isString())
381  theSelectionBox->setCurrentText(theValue.toString());
382  } else {
383 
384  // old file .. use defaults
385 
386 
387  numModesModal->setText("1");
388  theSelectionBox->setCurrentIndex(0);
389  theConvergenceTest->setText("NormUnbalance 1.0e-2 10");
390  theSolver->setText("Umfpack");
391  theAnalysis->setText("Transient -numSubLevels 2 -numSubSteps 10");
392  }
393 
394  if (jsonObject.contains("firstMode")) {
395  QJsonValue theValue = jsonObject["firstMode"];
396  if (theValue.isDouble())
397  firstMode->setText(QString::number(theValue.toInt()));
398  } else {
399  firstMode->setText("1"); // old code default
400  }
401 
402  if (jsonObject.contains("secondMode")) {
403  QJsonValue theValue = jsonObject["secondMode"];
404  if (theValue.isDouble())
405  secondMode->setText(QString::number(theValue.toInt()));
406  } else {
407  secondMode->setText("0"); // old code
408  }
409 
410  if (jsonObject.contains("numModesModal")) {
411  QJsonValue theValue = jsonObject["numModesModal"];
412  if (theValue.isDouble())
413  numModesModal->setText(QString::number(theValue.toInt()));
414  } else {
415  ; // old code
416  }
417 
418  if (jsonObject.contains("rayleighTangent")) {
419  QJsonValue theValue = jsonObject["rayleighTangent"];
420  if (theValue.isString())
421  theRayleighStiffness->setCurrentText(theValue.toString());
422  } else {
423  ; // old code
424  }
425 
426  if (jsonObject.contains("modalRayleighTangentRatio")) {
427  QJsonValue theValue = jsonObject["modalRayleighTangentRatio"];
428  if (theValue.isDouble())
429  dampingRatioModalTangent->setText(QString::number(theValue.toDouble()));
430  } else {
431  ; // old code
432  }
433 
434  if (jsonObject.contains("analysis")) {
435  QJsonValue theValue = jsonObject["analysis"];
436  theAnalysis->setText(theValue.toString());
437  } else {
438  theAnalysis->setText("Transient -numSubLevels 2 -numSubSteps 10");
439  ; // old code
440  }
441 
442  if (jsonObject.contains("solver")) {
443  QJsonValue theValue = jsonObject["solver"];
444  theSolver->setText(theValue.toString());
445  } else {
446  theSolver->setText("Umfpack");
447  ; // old code
448  }
449 
450  if (jsonObject.contains("fileName")) {
451  QJsonValue theName = jsonObject["fileName"];
452  QString fileName = theName.toString();
453  if (jsonObject.contains("filePath")) {
454  QJsonValue theName = jsonObject["filePath"];
455  QString filePath = theName.toString();
456  file->setText(QDir(filePath).filePath(fileName));
457  } else
458  return false;
459  }
460 
461 
462  return result;
