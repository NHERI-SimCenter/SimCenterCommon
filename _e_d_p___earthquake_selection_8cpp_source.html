166 bool
167 EDP_EarthquakeSelection::inputAppDataFromJSON(QJsonObject &jsonObject)
168 {
169 
170  // get name from "Application" key
171 
172  QString type;
173  if (jsonObject.contains("Application")) {
174  QJsonValue theName = jsonObject["Application"];
175  type = theName.toString();
176  } else {
177  emit sendErrorMessage("EDP_EarthquakeSelection - no Application key found");
178  return false;
179  }
180 
181  // based on application name value set edp type
182  int index = 0;
183  if ((type == QString("Standard Earthquake EDPs")) ||
184  (type == QString("StandardEarthquakeEDP"))) {
185  index = 0;
186  } else if ((type == QString("UserDefinedEDP")) ||
187  (type == QString("User Defined EDPs"))) {
188  index = 1;
189  } else {
190  emit sendErrorMessage("EDP_EarthquakeSelection - no valid type found");
191  return false;
192  }
193 
194  edpSelection->setCurrentIndex(index);
195 
196  // invoke inputAppDataFromJSON on new type
197 
198  if (theCurrentEDP != 0) {
199  return theCurrentEDP->inputAppDataFromJSON(jsonObject);
200  }
201 
202  return true;
203 }
