211 bool
212 GeneralInformationWidget::outputToJSON(QJsonObject &jsonObj){
213 
214  jsonObj["name"] = nameEdit->text().trimmed();
215  // jsonObj["revision"] = revEdit->text().toDouble();
216  // jsonObj["type"] = typeEdit->text().trimmed();
217  // jsonObj["year"] = yearEdit->text().toInt();
218  jsonObj["stories"] = storiesEdit->text().toInt();
219  jsonObj["width"] = widthEdit->text().toDouble();
220  jsonObj["depth"] = depthEdit->text().toDouble();
221 
222  // if the plan area is provided:
223  double planAreaValue = planAreaEdit->text().toDouble();
224  if (planAreaValue > 0.0)
225  jsonObj["planArea"] = planAreaValue;
226  else {
227  double widthValue = widthEdit->text().toDouble();
228  double depthValue = depthEdit->text().toDouble();
229  double rectAreaValue = widthValue * depthValue;
230  // otherwise, assume a rectangular plan and use width x depth
231  if (rectAreaValue > 0.0)
232  jsonObj["planArea"] = rectAreaValue;
233  else
234  // if none of the above are provided, use 0.0 for the area
235  jsonObj["planArea"] = 0.0;
236  }
237 
238  jsonObj["height"] = heightEdit->text().toDouble();
239  // jsonObj["weight"] = weightEdit->text().toDouble();
240 
241  QJsonObject location;
242 
243  /*
244  location["name"] = locationNameEdit->text().trimmed();
245  */
246 
247  QString dblVal = latitudeEdit->text();
248  location["latitude"] = dblVal.toDouble();
249 
250  dblVal = longitudeEdit->text();
251  location["longitude"] = dblVal.toDouble();
252 
253  jsonObj["location"] = location;
254 
255  QJsonObject units;
256  units["force"] = unitEnumToString(unitsForceCombo->currentData().value<ForceUnit>());
257  units["length"] = unitEnumToString(unitsLengthCombo->currentData().value<LengthUnit>());
258  units["time"] = unitEnumToString(unitsTimeCombo->currentData().value<TimeUnit>());
259  units["temperature"] = unitEnumToString(unitsTemperatureCombo->currentData().value<TemperatureUnit>());
260 
261  jsonObj["units"] = units;
262 
263  return true;
264 }

266 bool
267 GeneralInformationWidget::inputFromJSON(QJsonObject &jsonObject){
268  qDebug() << "General Information";
269 
270  QJsonValue nameValue = jsonObject["name"];
271  nameEdit->setText(nameValue.toString());
272 
273  QJsonValue storiesValue = jsonObject["stories"];
274  storiesEdit->setText(QString::number(storiesValue.toInt()));
275 
276  QJsonValue heightValue = jsonObject["height"];
277  if (!heightValue.isNull())
278  heightEdit->setText( QString::number(heightValue.toDouble()));
279  else
280  heightEdit->setText("0.0");
281 
282  QJsonValue widthValue = jsonObject["width"];
283  if (!widthValue.isNull())
284  widthEdit->setText( QString::number(widthValue.toDouble()));
285  else
286  widthEdit->setText("0.0");
287 
288  QJsonValue depthValue = jsonObject["depth"];
289  if (!depthValue.isNull())
290  depthEdit->setText( QString::number(depthValue.toDouble()));
291  else
292  depthEdit->setText("0.0");
293 
294  QJsonValue planAreaValue = jsonObject["planArea"];
295  if(planAreaValue.isUndefined() || planAreaValue == QJsonValue::Null || !planAreaValue.isDouble())
296  planAreaEdit->setText("0.0");
297  else
298  planAreaEdit->setText(QString::number(planAreaValue.toDouble()));
299 
300  // Location Object
301  QJsonValue locationValue = jsonObject["location"];
302  if (!locationValue.isNull()) {
303  QJsonObject locationObj = locationValue.toObject();
304 
305  QJsonValue locationLatitudeValue = locationObj["latitude"];
306  latitudeEdit->setText(QString::number(locationLatitudeValue.toDouble()));
307 
308  QJsonValue locationLongitudeValue = locationObj["longitude"];
309  longitudeEdit->setText(QString::number(locationLongitudeValue.toDouble()));
310  } else {
311  longitudeEdit->setText("0.0");
312  latitudeEdit->setText("0.0");
313  }
314 
315  // Units Object
316  QJsonValue unitsValue = jsonObject["units"];
317  QJsonObject unitsObj = unitsValue.toObject();
318 
319  QJsonValue unitsForceValue = unitsObj["force"];
320  ForceUnit forceUnit = unitStringToEnum<ForceUnit>(unitsForceValue.toString());
321  int forceUnitIndex = unitsForceCombo->findData(forceUnit);
322  unitsForceCombo->setCurrentIndex(forceUnitIndex);
323 
324  QJsonValue unitsLengthValue = unitsObj["length"];
325  LengthUnit lengthUnit = unitStringToEnum<LengthUnit>(unitsLengthValue.toString());
326  int lengthUnitIndex = unitsLengthCombo->findData(lengthUnit);
327  unitsLengthCombo->setCurrentIndex(lengthUnitIndex);
328 
329  QJsonValue unitsTimeValue = unitsObj["time"];
330  TimeUnit timeUnit = unitStringToEnum<TimeUnit>(unitsTimeValue.toString());
331  int timeUnitIndex = unitsTimeCombo->findData(timeUnit);
332  unitsTimeCombo->setCurrentIndex(timeUnitIndex);
333 
334  QJsonValue unitsTempValue = unitsObj["temperature"];
335  TemperatureUnit tempUnit = unitStringToEnum<TemperatureUnit>(unitsTempValue.toString());
336  int tempUnitIndex = unitsTemperatureCombo->findData(tempUnit);
337  unitsTemperatureCombo->setCurrentIndex(tempUnitIndex);
338 
339  return true;
340 }
