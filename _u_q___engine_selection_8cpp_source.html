141 UQ_EngineSelection::inputFromJSON(QJsonObject &jsonObject) {
142 
143  if (jsonObject.contains("UQ_Method")) {
144  QJsonObject uqObject = jsonObject["UQ_Method"].toObject();
145 
146  if (uqObject.contains("uqEngine")) {
147  QString engine =uqObject["uqEngine"].toString();
148  int index = theEngineSelectionBox->findText(engine);
149  if (index == -1) {
150  qDebug() << "UQ_EngineSelection::inputFromJson - Unknown Engine : " << engine;
151  return false;
152  }
153  theEngineSelectionBox->setCurrentIndex(index);
154  } else { // old format which just contained info for Dakota
155  theEngineSelectionBox->setCurrentIndex(0);
156  }
157 
158  if (theCurrentEngine != 0) {
159  return theCurrentEngine->inputFromJSON(uqObject);
160  }
161  }
162 
163  return false;
164 }

209 bool
210 UQ_EngineSelection::inputAppDataFromJSON(QJsonObject &jsonObject)
211 {
212  // get name from "Application" key
213 
214  if (jsonObject.contains("UQ")) {
215  QJsonObject theObject = jsonObject["UQ"].toObject();
216 
217  QString type;
218 
219  if (theObject.contains("Application")) {
220  QJsonValue theName = theObject["Application"];
221  type = theName.toString();
222 
223  // based on application name value set engine type
224  int index = 0;
225  if ((type == QString("Dakota")) ||
226  (type == QString("DakotaEngine")) ||
227  (type == QString("Dakota-UQ1")) ||
228  (type == QString("Dakota-UQ"))) {
229  index = 0;
230  } else if ((type == QString("UQpy")) || (type == QString("UQpyEngine"))) {
231  index = 1;
232  } else {
233  emit sendErrorMessage("UQ_EngineSelection - no valid type found");
234  return false;
235  }
236 
237  theEngineSelectionBox->setCurrentIndex(index);
238 
239  // invoke inputAppDataFromJSON on new type
240  if (theCurrentEngine != 0) {
241  return theCurrentEngine->inputAppDataFromJSON(theObject);
242  }
243  } else {
244  emit sendErrorMessage("UQ_EngineSelection - no Application key found");
245  return false;
246  }
247 
248  } else {
249  emit sendErrorMessage("UQ_EngineSelection: failed to find UQ application");
250  return false;
251  }
252 
253 
254  return false;
255 }
